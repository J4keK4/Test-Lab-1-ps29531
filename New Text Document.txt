import random

def main()
    number = random.randint(1, 100)
    guess = -1

    while guess != number
        guess = int(input(Nhập số ))

        if guess  number
            print(Số của bạn nhỏ hơn số cần đoán.)
        elif guess  number
            print(Số của bạn lớn hơn số cần đoán.)
        else
            print(Chúc mừng, bạn đã đoán đúng!)

if __name__ == __main__
    main()
